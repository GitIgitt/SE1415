\paragraph{FR001}
\begin{description}
     \item[Aussage]
        \textit{G\"aste sollen sich bestehende, \"offentliche Daten und Analysen bzw. Visualisierung dieser Daten ansehen k\"onnen.}
    \item [User requirement] \textit{UR018}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR002}
\begin{description}
     \item[Aussage]
        \textit{Registrierte Nutzer sollen neue Daten in das System per CSV-Datei hochladen k\"onnen. Mehrere Datens\"atze sollen parallel hochgeladen werden k\"onnen.}
    \item [User requirement] \textit{UR019}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR003}
\begin{description}
     \item[Aussage]
        \textit{Auf von ihnen hochgeladene Daten haben Nutzer automatisch Schreibrechte. Ausserdem sollen sie für die diese Datens\"atze Rechte an andere Nutzer vergeben k\"onnen.}
    \item [User requirement] \textit{UR021}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR004}
\begin{description}
     \item[Aussage]
        \textit{Registrierte Nutzer sollen von ihnen hochgeladene Datens\"atze l\"oschen k\"onnen.}
    \item [User requirement] \textit{UR019}
    \item [Priorit\"at] \textit{B}
\end{description}

\paragraph{FR005}
\begin{description}
     \item[Aussage]
        \textit{Registrierte Nutzer sollen vorhandene Daten analysieren und vergleichen k\"onnen, sofern sie dazu Rechte besitzen.}
    \item [User requirement] \textit{UR019}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR006}
\begin{description}
     \item[Aussage] \textit{Registrierte Nutzer sollen Nutzergruppen erstellen k\"onnen, um die Rechtevergabe auf von ihnen hochgeladene Daten zu erleichtern.}
    \item [User requirement] \textit{UR021}
    \item [Priorit\"at] \textit{B}
\end{description}

\paragraph{FR007}
\begin{description}
     \item[Aussage]
        \textit{Registrierte Nutzer sollen von ihnen hochgeladene Daten ver\"offentlichen k\"onnen, sodass sie auch von G\"asten und anderen Nutzern eingesehen werden k\"onnen, ohne dass sie die Rechte dazu vom hochladenden Nutzer einholen m\"ussen.}
    \item [User requirement] \textit{UR022}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR008}
\begin{description}
     \item[Aussage]
        \textit{Administratoren sollen die M\"oglichkeit haben, alle im System vorhandenen Datens\"atze zu analysieren, dessen Rechte zu verwalten und zu l\"oschen.}
    \item [User requirement] \textit{UR022}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR009}
\begin{description}
     \item[Aussage]
        \textit{Administratoren sollen registrierte Nutzer zu Administratoren bef\"ordern k\"onnen. Dabei sollen sie NICHT die M\"oglichkeit haben, anderen Administratoren die Administratoren-Rechte zu entziehen.}
    \item [User requirement] \textit{UR022}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR010}
\begin{description}
     \item[Aussage]
        \textit{Nutzer mit entsprechenden Rechten sollen die M\"oglichkeit haben, das Ergebnis einer Analyse von Daten und deren Visualisierung im PDF-Format herunterladen zu k\"onnen.}
    \item [User requirement] \textit{UR004}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR011}
\begin{description}
     \item[Aussage]
        \textit{Analysierte Daten sollen visualisiert werden k\"onnen. Als Visualisierungsm\"oglichkeiten sollen zur Verf\"ugung stehen:}
        \begin{itemize}
	      \item Balkendiagramm
	      \item Tortendiagramm
	      \item S\"aulendiagramm
	      \item Liniendiagramm
	      \item Ringdiagramm
   	   \end{itemize}
    \item [User requirement] \textit{UR022}
    \item [Priorit\"at] \textit{A}
\end{description}

\paragraph{FR010}
\begin{description}
     \item[Aussage]
        \textit{Der Nutzer soll die M\"oglichkeit haben, einen ganzen Datensatz zu visualisieren oder einen von ihm definierten Ausschnitt dessen.}
    \item [User requirement] \textit{UR004}
    \item [Priorit\"at] \textit{B}
\end{description}
